type User = {
  id: string
  name: string
  age: number
}

function omit <T, K extends keyof T>(entity: T, propertyToOmit: K) {
  const {[propertyToOmit]: omitted, ...rest} = entity

  return rest
}

const user: User = {
    id: 'uuid',
    name: 'RainbowDash',
    age: 23,
}

const userWithoutName = omit(user, 'name')
