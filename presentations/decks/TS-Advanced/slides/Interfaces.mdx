# Interfaces

```tsx {all|1|3-7|9-14}
enum DraggingStatus { Untouched, Dragging, Dropped }

interface Draggable {
    draggingStatus: DraggingStatus
    onDrag: () => void
    onDrop?: (e: DragEvent) => void
}

class Box implements Draggable {
  draggingStatus: DraggingStatus

  constructor() {
      this.draggingStatus = DraggingStatus.Untouched
  }

  onDrag = () => {
      this.draggingStatus = DraggingStatus.Dragging
  }
}

const box = new Box()

box.onDrag()
console.log(box.draggingStatus)
```
