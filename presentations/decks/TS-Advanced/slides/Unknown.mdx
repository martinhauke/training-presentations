const ama = (me: unknown): string => {
  if (typeof me === 'number') {
    return 'IAMA number, ask me anything!'
  }

  if (typeof me === 'string') {
    return `IAMA ${me.length} characters long string, ask me anything!`
  }

  return 'I don\'t really know, don\'t ask'
}

console.log(ama(23))
console.log(ama('cool'))
console.log(ama(true))
