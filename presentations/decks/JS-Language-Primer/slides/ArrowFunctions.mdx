```tsx title="Arrow Functions"
// basic function declaration
function double(x) {
  return 2 * x
}

// equivalent arrow function
const double = (x) => {
  return 2 * x
}
```

```tsx title="Arrow Functions"
// even shorter
// -> single argument needs
//    no parantheses
// -> expressions can be
//    returned implicitly
//    (without func body and "return")
const double = x => 2 * x
```
