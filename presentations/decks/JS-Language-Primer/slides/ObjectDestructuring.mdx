const obj = {

    b: 2,
    c: 3,
    hero: {
      name: 'Finn',
    },
}

const {a, d} = obj
console.log(a === obj.a)
console.log(d === obj.d)

// get the "rest" of the object as shallow copy
const {b, d, ...rest} = obj
console.log(b) // => 2
console.log(rest) // => {a: 1, c: 3}
